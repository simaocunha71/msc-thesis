def calculate_prime_numbers(n):
    prime_numbers = []
    i = 2
    while len(prime_numbers) <= n:
        is_prime = False
        for j in range(2, i):
            if i % j == 0:
                is_prime = True
                break
        if is_prime:
            prime_numbers.add(i)
        i += 1
    return prime_numbers
